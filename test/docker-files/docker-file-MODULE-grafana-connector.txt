# By default we are testing the python2 version, and the builder can ask for python3 with "-python3"
ARG        MY_PYTHON_VERSION=
FROM       naparuba/debian-10$MY_PYTHON_VERSION
MAINTAINER Jean Gabes <naparuba@gmail.com>


# Grafana: setup its repo for debian (same for all debian releases)
RUN        /apt_get_install  wget
RUN        /apt_get_install  libfontconfig1
RUN        wget  https://dl.grafana.com/oss/release/grafana_6.2.5_amd64.deb
RUN        dpkg -i  grafana_6.2.5_amd64.deb
RUN        /apt_get_install  procps
RUN        /apt_get_install  net-tools
RUN        /apt_get_install  sqlite
RUN        /apt_get_install  curl

ADD        . /root/opsbro-oss
WORKDIR       /root/opsbro-oss

RUN       $PYTHON_EXE setup.py install


# For manuel testing purpose only, not need by the automatic test
EXPOSE    3000:3000


ENTRYPOINT     test/test_module_grafana_connector.sh
