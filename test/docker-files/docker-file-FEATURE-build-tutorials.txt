# By default we are testing the python2 version, and the builder can ask for python3 with "-python3"
ARG        MY_PYTHON_VERSION=
FROM       naparuba/debian-9$MY_PYTHON_VERSION
MAINTAINER Jean Gabes <naparuba@gmail.com>

ADD        test/docker-helper/  /

RUN        /apt_get_install   locales
RUN        /apt_get_install   python3-pkg-resources
RUN        /apt_get_install   asciinema

# Locales
RUN        cp /etc/locale.gen /etc/locale.gen.old
RUN        sed -i "s/# en_US.UTF-8 UTF-8/en_US.UTF-8 UTF-8/g" /etc/locale.gen
RUN        /usr/sbin/locale-gen

ENV        LANG=en_US.UTF8


ADD        . /root/opsbro-oss
WORKDIR       /root/opsbro-oss


ENTRYPOINT     /bin/bash test/feature_build_tutorials.sh
