# By default we are testing the python2 version, and the builder can ask for python3 with "-python3"
ARG        MY_PYTHON_VERSION=
FROM       naparuba/debian-10$MY_PYTHON_VERSION
MAINTAINER Jean Gabes <naparuba@gmail.com>



# HTTP node
RUN        /apt_get_install   procps
RUN        /apt_get_install   curl
RUN        /apt_get_install   apache2
RUN        echo "NODE-HTTP-2" > /var/www/html/index.html

ADD        . /root/opsbro-oss
WORKDIR       /root/opsbro-oss

RUN       $PYTHON_EXE setup.py install

# Ask for an encrypted test
RUN       opsbro gossip zone key import --zone internet --key "NGNjZWI2ZmEyMzEyMTFlOA=="

# The node1 will try to connect to node2 and auto join it
ENTRYPOINT    /etc/init.d/apache2 start; test/test_demo1_http.sh "NODE-HTTP-2"
