# By default we are testing the python2 version, and the builder can ask for python3 with "-python3"
ARG        MY_PYTHON_VERSION=
FROM       naparuba/debian-10$MY_PYTHON_VERSION
MAINTAINER Jean Gabes <naparuba@gmail.com>

ADD        test/docker-helper/  /


RUN        /apt_get_install   python-jinja2
RUN        /apt_get_install    python-crypto
RUN        /apt_get_install    python-setuptools
RUN        /apt_get_install    python-leveldb
RUN        /apt_get_install    python-simplejson
RUN        /apt_get_install    python-setproctitle
RUN        /apt_get_install    python-psutil

ADD        . /root/opsbro-oss

WORKDIR    /root/opsbro-oss

RUN        python setup.py install

# Copy the generator
ADD        test/test-files/test-generators/   /var/lib/opsbro/global-configuration/packs/
RUN        ls -R /var/lib/opsbro/global-configuration/packs/test-generators/
RUN        cp   /var/lib/opsbro/global-configuration/packs/test-generators/generators/templates/authorized_keys.txt  /tmp/

RUN        opsbro agent parameters add groups ssh-key-sample

ENTRYPOINT     test/test_feature_generators.sh
